# Coloca o branch na linha de comando e colore.
export PS1='\w$(__git_ps1 " \[\033[0;31m\](%s)\[\033[0m\]") \$ '

# Define qual o editor. Ex.: 'mate -w'
export EDITOR='subl -w'

# Aliases
alias ls="ls -G"
alias r="rails"

# Metodos
# EACH - Metodo para executar algum comando para todas as pastas. Ex.: each git pull
each () {
 for f in `ls`; do
   echo "--> $f"
   cd $f
   $*
   cd ..
 done
}
